`timescale 1ns / 1ps

module DAY65UPDOWNCOUNTER(
    input clk,reset,updown,
    output reg [3:0]counter
    );
    always @(posedge clk)
    if(reset)
        counter <= 0;
    else if(updown)
        if(counter == 15)
            counter <= 0;
        else
            counter <= counter+1;
    else
        if(counter == 0)
            counter <= 15;
        else
            counter <= counter-1;
endmodule
