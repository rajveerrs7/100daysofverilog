`timescale 1ns / 1ps
module DAY47SRLATCH(
    input en,reset,s,r,
    output reg q
    );
    always @(*)
    begin
        if(reset)
            q<=1'b0;
        else if(en) begin
            case({s,r})
                2'b00:q<=q;
                2'b01:q<=1'b0;
                2'b10:q<=1'b1;
                default:q<=1'bx;
            endcase
        end
    end
endmodule
