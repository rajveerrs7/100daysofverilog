`timescale 1ns / 1ps
module DAY15MULTIPLIER(
input [3:0]a,b,
output reg [7:0]out
    );
    reg [7:0]r0,r1,r2,r3;
    always @(a,b)
    begin
    r0 = 8'b0;
    r1 = 8'b0;
    r2 = 8'b0;
    r3 = 8'b0;

    if(b[0])
        r0=a;
    if(b[1])
        r1=a<<1;
    if(b[2])
        r2=a<<2;
    if(b[3])
        r3=a<<3;
    
    out=r0+r1+r2+r3;
    end
endmodule
