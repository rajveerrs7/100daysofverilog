`timescale 1ns / 1ps

module DAY82ROM
    #(parameter
    addr_width = 4,
    data_width = 16,
    depth = 16
    )
    (
    input clk, en,
    input [addr_width-1:0]addr,
    output [data_width-1:0]data
    );
    
    reg [data_width-1:0]mem[depth-1:0];
    reg addr_reg;
    
    always @(posedge clk)
        if(en)
            addr_reg <= addr;
    
    assign data = mem[addr_reg];
    
endmodule
