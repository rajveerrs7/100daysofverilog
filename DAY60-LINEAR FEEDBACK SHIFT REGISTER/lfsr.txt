`timescale 1ns / 1ps

module DAY60LFSR(
    input clk, reset,
    output [7:0]LFSR
    );
    reg [7:0]data;
    always @(posedge clk)begin
        if(reset)
            data <= 8'b11111111;
        else
            data <= {data[6:0],data[7]^data[5]^data[3]^data[1]};
    end
    assign LFSR = data;
endmodule
