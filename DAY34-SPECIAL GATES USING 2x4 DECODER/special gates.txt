`timescale 1ns / 1ps
module DAY34SPECIALGATES(
input a,b,
output out_xor,out_xnor
    );
    wire [3:0]w;
    DECODER4x2 decoder({a,b},w);
    assign out_xor=w[1]|w[2];
    assign out_xnor=w[3]|w[0];
endmodule


module DECODER4x2(
input [1:0]in,
output reg [3:0]out
);
always @(in)
    begin
        case(in)
            2'b00:out=4'b0001;
            2'b01:out=4'b0010;
            2'b10:out=4'b0100;
            2'b11:out=4'b1000;
        endcase
    end
endmodule
