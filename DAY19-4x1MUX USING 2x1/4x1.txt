`timescale 1ns / 1ps
module DAY19MUX(
input [3:0]a,
input [1:0]s,
output out
    );
    wire [1:0]w;
    MUX2x1 mux1(a[1:0],s[0],w[0]);
    MUX2x1 mux2(a[3:2],s[0],w[1]);
    MUX2x1 mux3(w,s[1],out);
endmodule

module MUX2x1(
input [1:0]in,
input sel,
output out
);
assign out=sel?in[1]:in[0];
endmodule
