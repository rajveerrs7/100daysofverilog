`timescale 1ns / 1ps

module DAY63NBITCOUNTER(
    clk,reset,counter
    );
    parameter N=3, LENGTH=6;
    input clk,reset;
    output reg [N-1:0]counter;
    
    always @(posedge clk)begin
        if(reset)
            counter <= 0;
        else begin
            if(counter == LENGTH-1)
                counter <= 0;
            else
                counter <= counter+1;
        end
    end
endmodule
