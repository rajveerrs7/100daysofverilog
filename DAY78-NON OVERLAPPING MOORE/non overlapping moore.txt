`timescale 1ns / 1ps

module DAY78FSM(
    input clk, reset, in, 
    output reg [3:0]state,
    output out
    );
    
    reg [3:0]next_state;
    parameter [3:0]IDLE   = 4'b0001,
                   STATE1 = 4'b0010,
                   STATE2 = 4'b0011,
                   STATE3 = 4'b0100,
                   STATE4 = 4'b0101;
                   
    always @(*)
        begin
            case(state)
                IDLE : begin
                    if(in)begin
                        next_state <= STATE1;
                    end
                    else begin
                        next_state <= IDLE;
                    end
                end
                STATE1 : begin
                    if(~in)begin
                        next_state <= STATE2;
                    end
                    else begin
                        next_state <= IDLE;
                    end
                end
                STATE2 : begin
                    if(in)begin
                        next_state <= STATE3;
                    end
                    else begin
                        next_state <= IDLE;
                    end
                end
                STATE3 : begin
                    if(in)begin
                        next_state <= STATE4;
                    end
                    else begin
                        next_state <= IDLE;
                    end
                end
                STATE4 : begin
                    if(in)begin
                        next_state <= STATE1;
                    end
                    else begin
                        next_state <= IDLE;
                    end
                end
                default : next_state <= IDLE;
            endcase
        end
        
        always @(posedge clk)
        if(reset)
            state <= 4'b0000;
        else
            state <= next_state;
        
        assign out = (state == STATE4)? 1:0;
        
endmodule
